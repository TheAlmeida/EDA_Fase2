21double calculateDistance(double lat1, double lon1, double lat2, double lon2) {
22 double dLat = degreesToRadians(lat2 - lat1);
23 double dLon = degreesToRadians(lon2 - lon1);
24
25 double a = sin(dLat / 2) * sin(dLat / 2) +
26 cos(degreesToRadians(lat1)) * cos(degreesToRadians(lat2)) *
27 sin(dLon / 2) * sin(dLon / 2);
28 double c = 2 * atan2(sqrt(a), sqrt(1 - a));

30 double distance = EARTH_RADIUS * c;
